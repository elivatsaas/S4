TEST_CASE("Testing the timeToNumber function")
{
    string time1, time2, time3, time4;
    time1 = "1000";
    time2 = "1234";
    time3 = "0000";
    time4 = "1859";
    CHECK(timeToNumber(time1) == 600);
    CHECK(timeToNumber(time2) == 754);
    CHECK(timeToNumber(time3) == 0);
    CHECK(timeToNumber(time4) == 1139);
}
